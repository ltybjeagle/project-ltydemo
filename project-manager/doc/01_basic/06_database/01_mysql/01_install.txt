1、检查库文件是否存在，如有删除：
	[root@slave ~]# rpm -qa | grep mysql
	mysql-libs-5.1.71-1.el6.x86_64
	[root@slave ~]# rpm -e mysql-libs-5.1.71-1.el6.x86_64 --nodeps
	[root@slave ~]# rpm -qa | grep mysql
2、检查mysql组和用户是否存在，如无创建：
	[root@slave ~]# cat /etc/group | grep mysql
	[root@slave ~]# cat /etc/passwd | grep mysql
	// useradd -r参数表示mysql用户是系统用户，不可用于登录系统。
	[root@slave ~]# groupadd mysql
	[root@slave ~]# useradd -r -g mysql mysql
3、解压TAR包，更改所属的组和用户：
	[root@slave zkpk]# cd /usr/local/
	[root@slave local]# tar -zxvf mysql-5.7.14-linux-glibc2.5-x86_64.tar.gz 
	[root@slave local]# mv mysql-5.7.14-linux-glibc2.5-x86_64 mysql
	[root@slave local]# chown -R mysql:mysql mysql/
	[root@slave local]# cd mysql/
3、配置OS：
	[root@slave local]# vim /etc/security/limits.conf
		mysql hard nofile 65535
		mysql soft nofile 65535
4、配置my.cnf文件
	[root@slave local]# rm -rf /etc/my.cnf
	[root@slave local]# vim /etc/my.cnf
		#此文件内容如下(路径根据自己的实际情况):
		[client]
		port = 3306
		socket = /tmp/mysql.sock

		[mysqld]
		init-connect='SET NAMES utf8'
		basedir=/usr/local/mysql          #根据自己的安装目录填写 
		datadir=/usr/local/mysql/data     #根据自己的mysql数据目录填写
		socket=/tmp/mysql.sock
		max_connections=200               # 允许最大连接数
		character-set-server=utf8         # 服务端使用的字符集默认为8比特编码的latin1字符集
		default-storage-engine=INNODB     # 创建新表时将使用的默认存储引擎
5、安装与初始化DB：
	[root@slave mysql]# bin/mysqld --initialize-insecure --user=mysql --basedir=/usr/local/mysql/ --datadir=/usr/local/mysql/data/
	[root@slave mysql]# chown -R root:root /usr/local/mysql/
	[root@slave mysql]# chown -R mysql:mysql /usr/local/mysql/data/
6、修改root密码
	[root@slave mysql]# ./bin/mysql -u root -p     #默认没有密码,直接敲回车就可以
		mysql> use mysql;
		mysql> update user set authentication_string=password('123456') where user='root';
		mysql> flush privileges;
		mysql> exit;
7、设置为开机启动
	[root@slave mysql]# cp ./support-files/mysql.server  /etc/init.d/mysql
	[root@slave mysql]# chkconfig --add mysql
	[root@slave mysql]# chkconfig --list
	[root@slave mysql]# chkconfig --level 345 mysql on
8、添加远程访问权限
	[root@slave mysql]# ./bin/mysql -u root -p #输入密码登陆
		mysql> use mysql; 
		mysql> update user set host = '%' where user = 'root';
		mysql> select host, user from user;
