安装：
    [root@localhost ~]# cd /usr/local
    [root@localhost local]# mkdir zookeeper
    [root@localhost zookeeper]# tar -xzvf zookeeper-3.4.10.tar.gz
    [root@localhost zookeeper]# cd zookeeper-3.4.10/
    [root@localhost zookeeper-3.4.10]# cd conf/
    [root@localhost conf]# cp zoo_sample.cfg zoo.cfg
    [root@localhost conf]# vi zoo.cfg
    tickTime=2000 --follower向leader发送心跳的时间间隔，2秒心跳一次
    initLimit=10 --启动初始化的时间为10个心跳
    syncLimit=5 --leader向follower发送请求让follower同步数据，若5个心跳没有收到回应，就认为超时了
    clientPort=2181 --zookeeper接收客户端请求的端口号
    dataDir=/usr/local/zookeeper/zookeeper-3.4.6/data --保存数据目录
    dataLogDir=/usr/local/zookeeper/zookeeper-3.4.6/log --保存日志目录
    server.1=ip:2888:3888 --其中2888是leader和follower通信的端口，3888是选举通信的端口（ip为服务器的ip）
    [root@localhost zookeeper-3.4.10]# mkdir data
    [root@localhost zookeeper-3.4.10]# mkdir log
    [root@localhost zookeeper-3.4.10]# cd data
    [root@localhost data]# echo "1" > myid
    [root@localhost data]# vim /etc/profile
    ZOOKEEPER_HOME=/usr/local/zookeeper/zookeeper-3.4.10
    export ZOOKEEPER_HOME
    export PATH=${PATH}:${ZOOKEEPER_HOME}/bin
    [root@localhost bin]# cd /usr/local/zookeeper/zookeeper-3.4.6/bin
    [root@localhost bin]# sh zkServer.sh start --启动
    [root@localhost bin]# sh zkServer.sh status --查看zookeeper的启动状态
